\section{线性映射相关知识}
\begin{definition}{线性映射}
    设 \(V\) 和 \(U\) 是数域 \(K\) 上的线性空间，\(\varphi\) 是 \(V\) 到 \(U\) 的映射，如果对任意 \(\alpha, \beta \in V\) 和任意数 \(k\)，都有
    \[
    \varphi(\alpha + \beta) = \varphi(\alpha) + \varphi(\beta), \quad \varphi(k\alpha) = k\varphi(\alpha),
    \]
    则称 \(\varphi\) 是 \(V\) 到 \(U\) 的线性映射。
\end{definition}

\begin{definition}{线性变换}
    设 \(V\) 是数域 \(K\) 上的线性空间，\(\varphi\) 是 \(V\) 到 \(V\) 的映射，如果 \(\varphi\) 是 \(V\) 到 \(V\) 的线性映射，则称 \(\varphi\) 是 \(V\) 上的线性变换。
\end{definition}

\begin{definition}{像空间}
    设 \(\varphi\) 是线性空间 \(V\) 到 \(U\) 的线性映射，\(\varphi(V) = \{\varphi(\alpha): \alpha \in V\}\)，称 \(\varphi(V)\) 是线性映射 \(\varphi\) 的像空间，记作 \(Im\varphi\)。
\end{definition}

\begin{definition}{核空间}
    设 \(\varphi\) 是线性空间 \(V\) 到 \(U\) 的线性映射，\(Ker\varphi = \{\alpha \in V: \varphi(\alpha) = 0\}\)，称 \(Ker\varphi\) 是线性映射 \(\varphi\) 的核空间，记作 \(Ker\varphi\)。
\end{definition}

\begin{theorem}
    设 \(\varphi\) 是数域 \(\mathbb{F}\) 上向量空间 \(V\) 到 \(U\) 的线性映射，\(A\) 是 \(\varphi\) 在任意给定基下的表示矩阵，则 \(\varphi\) 的秩等于 \(A\) 的秩，\(\varphi\) 的零度等于 \(V\) 的维数减去 \(A\) 的秩。即有
\[
\dim \operatorname{Im} \varphi + \dim \operatorname{Ker} \varphi = \dim V.
\]
\end{theorem}

\begin{definition}{不变子空间}
    设 \(\varphi\) 是线性空间 \(V\) 上的线性变换，如果 \(V\) 的子空间 \(W\) 满足 \(\varphi(W) \subseteq W\)，则称 \(W\) 是线性变换 \(\varphi\) 的不变子空间。
\end{definition}

\begin{theorem}
    设 \(\varphi\) 是数域 \(\mathbb{F}\) 上向量空间 \(V\) 上的线性变换，\(W\) 是 \(\varphi\) 的不变子空间。若取 \(W\) 的一组基 \(\{e_1, \cdots, e_r\}\)，再扩张为 \(V\) 的一组基 \(\{e_1, \cdots, e_r, e_{r+1}, \cdots, e_n\}\)，则 \(\varphi\) 在这组基下的表示矩阵具有下列分块上三角矩阵的形状：
\[
\begin{pmatrix}
A_{11} & A_{12} \\
O & A_{22}
\end{pmatrix},
\]
其中 \(A_{11}\) 是一个 \(r\) 阶矩阵。
\end{theorem}


\section{习题讲解}

\begin{example}\label{ex4.8}
    设 \(\varphi\) 是 \(n\) 维线性空间 \(V\) 上的线性变换，\(\alpha \in V\)。若 \(\varphi^{m-1}(\alpha) \neq 0\)，而  
    \(\varphi^m(\alpha) = 0\)，求证：\(\alpha, \varphi(\alpha), \varphi^2(\alpha), \cdots, \varphi^{m-1}(\alpha)\) 线性无关。
\end{example}
    
\begin{proof}
    设有 \(m\) 个数 \(a_0, a_1, \cdots, a_{m-1}\)，使得  
    \[
    a_0\alpha + a_1\varphi(\alpha) + \cdots + a_{m-1}\varphi^{m-1}(\alpha) = 0.
    \]
    
    上式两边同时作用 \(\varphi^{m-1}\)，则有  
    \[
    a_0\varphi^{m-1}(\alpha) = 0,
    \]
    由于 \(\varphi^{m-1}(\alpha) \neq 0\)，故 \(a_0 = 0\)。上式两边同时作用 \(\varphi^{m-2}\)，则有  
    \[
    a_1\varphi^{m-1}(\alpha) = 0,
    \]
    由于 \(\varphi^{m-1}(\alpha) \neq 0\)，故 \(a_1 = 0\)。不断这样做下去，最后可得  
    \[
    a_0 = a_1 = \cdots = a_{m-1} = 0,
    \]
    于是 \(\alpha, \varphi(\alpha), \varphi^2(\alpha), \cdots, \varphi^{m-1}(\alpha)\) 线性无关。
\end{proof}

\begin{example}
    设 \( V \) 是数域 \( K \) 上的 \( n \) 维线性空间，\( \varphi \) 是 \( V \) 上的幂零线性变换，满足  
    \( r(\varphi) = n - 1 \)。求证：存在 \( V \) 的一组基，使得 \( \varphi \) 在这组基下的表示矩阵为  
    \[
    A = 
    \begin{pmatrix}
    0 & 0 & \cdots & 0 \\
    1 & 0 & \cdots & 0 \\
    0 & 1 & \cdots & 0 \\
    \vdots & \vdots & \ddots & \vdots \\
    0 & 0 & \cdots & 1
    \end{pmatrix}.
    \]
\end{example}
    
\begin{proof}
    由假设存在正整数 \( m \)，使得 \( \varphi^m = 0, \varphi^{m-1} \neq 0 \)，从而存在 \( \alpha \in V \)，使得  
    \[
    \varphi^m(\alpha) = 0, \quad \varphi^{m-1}(\alpha) \neq 0。
    \]  
    由例\ref{ex4.8}可知，\( \alpha, \varphi(\alpha), \cdots, \varphi^{m-1}(\alpha) \) 线性无关，于是  
    \[
    m \leq \dim V = n。
    \]  
    另一方面，由 Sylvester 不等式$r(AB) \geq r(A) + r(B) -n $以及  
    \[
    r(\varphi) = n - 1
    \]  
    可知，
    \[
    r(\varphi^2) \geq 2r(\varphi) - n = n - 2。
    \]  
    不断这样讨论下去，最终可得  
    \[
    0 = r(\varphi^m) \geq n - m,
    \]  
    即有  
    \[
    m \geq n,
    \]  
    从而  
    \[
    m = n。
    \]  
    于是 \( \alpha, \varphi(\alpha), \cdots, \varphi^{n-1}(\alpha) \) 是 \( V \) 的一组基，\( \varphi \) 在这组基下的表示矩阵为 \( A \)。
\end{proof}

\begin{example}
    设 \(\varphi\) 是 \(n\) 维线性空间 \(V\) 上的线性变换，证明：\(\varphi\) 是可逆变换的充要条件是 \(\varphi\) 将 \(V\) 的基变为基。
\end{example}
    
\begin{proof}
若 \(\varphi\) 是可逆变换，则显然 \(\varphi\) 将 \(V\) 的基变为基。反之，若 \(e_1, e_2, \cdots, e_n\) 和 \(f_1, f_2, \cdots, f_n\) 是 \(V\) 的两组基，使得 \(\varphi(e_i) = f_i \ (1 \leq i \leq n)\)，则对任意 \(\alpha \in V\)，
\[
\alpha = \lambda_1 f_1 + \lambda_2 f_2 + \cdots + \lambda_n f_n,
\]
有 \(\varphi(\lambda_1 e_1 + \lambda_2 e_2 + \cdots + \lambda_n e_n) = \alpha\)，即 \(\varphi\) 是满映射，从而是自同构。我们也可以证明 \(\varphi\) 是单映射，从而是自同构（留给读者完成）。另外还可以这样讨论，设从基 \(e_1, e_2, \cdots, e_n\) 到基 \(f_1, f_2, \cdots, f_n\) 的过渡矩阵为 \(P\)，则 \(\varphi\) 在基 \(e_1, e_2, \cdots, e_n\) 下的表示矩阵就是 \(P\)，这是一个可逆矩阵，从而 \(\varphi\) 是可逆变换。
\end{proof}

\begin{example}\label{ex4.21}
    设 \(\varphi\) 是线性空间 \(V\) 到 \(U\) 的线性映射，\(\{e_1, e_2, \cdots, e_n\}\) 和 \(\{f_1, f_2, \cdots, f_n\}\) 是 \(V\) 的两组基，\(\{e_1, e_2, \cdots, e_n\}\) 到 \(\{f_1, f_2, \cdots, f_n\}\) 的过渡矩阵为 \(P\)。\(\{g_1, g_2, \cdots, g_m\}\) 和 \(\{h_1, h_2, \cdots, h_m\}\) 是 \(U\) 的两组基，\(\{g_1, g_2, \cdots, g_m\}\) 到 \(\{h_1, h_2, \cdots, h_m\}\) 的过渡矩阵为 \(Q\)。又设 \(\varphi\) 在基 \(\{e_1, e_2, \cdots, e_n\}\) 和基 \(\{g_1, g_2, \cdots, g_m\}\) 下的表示矩阵为 \(A\)，在基 \(\{f_1, f_2, \cdots, f_n\}\) 和基 \(\{h_1, h_2, \cdots, h_m\}\) 下的表示矩阵为 \(B\)。求证：
    \[
    B = Q^{-1}AP.
    \]
\end{example}
    
\begin{proof}
    任取 \(v \in V\)，设它在基 \(\{e_1, e_2, \cdots, e_n\}\) 下的坐标向量为 \((x_1, x_2, \cdots, x_n)'\)，则它在基 \(\{f_1, f_2, \cdots, f_n\}\) 下的坐标向量为 \(P^{-1}(x_1, x_2, \cdots, x_n)'\)。\(\varphi(v)\) 在基 \(\{g_1, g_2, \cdots, g_m\}\) 下的坐标向量为 \(A(x_1, x_2, \cdots, x_n)'\)，在基 \(\{h_1, h_2, \cdots, h_m\}\) 下的坐标向量为 \(BP^{-1}(x_1, x_2, \cdots, x_n)'\)。由于从 \(\{g_1, g_2, \cdots, g_m\}\) 到 \(\{h_1, h_2, \cdots, h_m\}\) 的过渡矩阵
    为 \(Q\)，故 \(A(x_1, x_2, \cdots, x_n)' = QBP^{-1}(x_1, x_2, \cdots, x_n)'\)。因为 \((x_1, x_2, \cdots, x_n)'\) 是任意的，故 \(A = QBP^{-1}\)，即 \(B = Q^{-1}AP\)。
\end{proof}


\begin{example}\label{ex4.22}
    设 \(\varphi\) 是有限维线性空间 \(V\) 到 \(U\) 的线性映射，求证：必存在 \(V\) 和 \(U\) 的两组基，使线性映射 \(\varphi\) 在两组基下的表示矩阵为  
    \[
    \begin{pmatrix} 
    I_r & O \\ 
    O & O 
    \end{pmatrix}.
    \]
\end{example}
    
\begin{proof}
    设 \(\{e_1, e_2, \cdots, e_n\}\) 是 \(V\) 的一组基，\(\{g_1, g_2, \cdots, g_m\}\) 是 \(U\) 的一组基，\(\varphi\) 在这两组基下的表示矩阵为 \(A\)。由相抵标准型理论可知，存在 \(m\) 阶非异阵 \(Q, n\) 阶非异阵 \(P\)，使得  
    \[
    Q^{-1}AP = 
    \begin{pmatrix} 
    I_r & O \\ 
    O & O 
    \end{pmatrix}.
    \]  
    设 \(\{f_1, f_2, \cdots, f_n\}\) 是 \(V\) 的一组新基，使得从 \(\{e_1, e_2, \cdots, e_n\}\) 到 \(\{f_1, f_2, \cdots, f_n\}\) 的过渡矩阵为 \(P\)；设 \(\{h_1, h_2, \cdots, h_m\}\) 是 \(U\) 的一组新基，使得从 \(\{g_1, g_2, \cdots, g_m\}\) 到 \(\{h_1, h_2, \cdots, h_m\}\) 的过渡矩阵为 \(Q\)，则由例 \ref{ex4.21} 可知，\(\varphi\) 在两组新基下的表示矩阵为  
    \[
    Q^{-1}AP = 
    \begin{pmatrix} 
    I_r & O \\ 
    O & O 
    \end{pmatrix}.
    \]
\end{proof}

\begin{example}

\end{example}
    
\begin{proof}
    取定 \(V\) 的一组基，设 \(\varphi\) 在这组基下的表示矩阵是 \(A\)。由已知条件可知，对任意一个同阶可逆矩阵 \(P\)，\(A = P^{-1}AP\)，即 \(PA = AP\)。因此矩阵 \(A\) 和任意一个可逆矩阵乘法可交换，于是 \(A = kI_n\)，由此即知 \(\varphi\) 是纯量变换。
\end{proof}

\begin{example}
    用几何方法证明 Cayley-Hamilton定理：已知$\mathcal{A}  $是有限维线性空间V中的线性变换，\(f(\lambda)\) 是变换 \(\mathcal{A}\) 的特征多项式，则 \(f(\mathcal{A}) = 0\)。
\end{example}

\begin{proof}
    要证明 \(f(\mathcal{A}) = 0\)，只要证明 \(f(\mathcal{A}) \xi = 0\) 对任意 \(\xi \in V\) 都成立即可。 \par
    任取 $\xi  \in V$ 若$\xi = 0$ ，则$f(\mathcal{A}) \xi = 0$ 显然成立。 \par
    若$\xi \neq 0$ ，考虑$\mathcal{A}\xi$,若$\xi$与$\mathcal{A}\xi$ 线性相关，则停止，令$W = L(\xi)$，否则继续考虑$\mathcal{A}^2\xi$，若$\{\xi,\mathcal{A}\xi,\mathcal{A}^2\xi\}$ 线性相关，则停止，令$W = L(\xi,\mathcal{A}\xi)$，否则继续讨论下去。由于V是有限维线性空间，$dim V = n < \infty$，V中$k >n$个向量必定线性相关，上述讨论过程一定可以停止，所以不妨设存在正整数$m$使得$\{\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^{m-1}\xi\}$ 线性无关，而$\{\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^m\xi\}$ 线性相关。 \par
    所以$\mathcal{A}^m\xi$可由线性无关的向量组$\{\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^{m-1}\xi\}$ 线性表示，即存在数 $a_0, a_1, \cdots, a_{m-1}$ 使得
\[
\mathcal{A}^m\xi = a_0 \xi + a_1 \mathcal{A} \xi + \cdots + a_{m-1} \mathcal{A}^{m-1} \xi.
\]
    令$W = L(\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^{m-1}\xi)$，则W为不变子空间。将W中的一组基$\{\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^{m-1}\xi\}$，扩张成V中的一组基$\{\xi,\mathcal{A}\xi,\cdots,\mathcal{A}^{m-1}\xi,v_{m+1},\cdots,v_{n}\}$则$\mathcal{A}$在这组基下的表示矩阵为
\[
\begin{pmatrix}
A_{11} & A_{12} \\
O & A_{22}
\end{pmatrix}
\]
其中$A_{11}$为线性变换$\mathcal{A}$限制在W上的表示矩阵。且
\[
A_{11} = \begin{pmatrix}
0 & 0 & \cdots & 0 & a_0 \\
1 & 0 & \cdots & 0 & a_1 \\
0 & 1 & \cdots & 0 & a_2 \\
\vdots & \vdots & \ddots & \vdots & \vdots \\
0 & 0 & \cdots & 1 & a_{m-1}
\end{pmatrix}
\]

\[
f(\lambda) = \begin{vmatrix}
\lambda I_n - \begin{pmatrix}
A_{11} & A_{12} \\
O & A_{22}
\end{pmatrix}
\end{vmatrix} = \begin{vmatrix}
\lambda I_m - A_{11} & -A_{12} \\
O & \lambda I_{n-m} - A_{22}
\end{vmatrix} = \begin{vmatrix}
\lambda I_m - A_{11}
\end{vmatrix} \begin{vmatrix}
\lambda I_{n-m} - A_{22}
\end{vmatrix}
\]
令$g(\lambda) = \begin{vmatrix}
    \lambda I_m - A_{11}
    \end{vmatrix}$，
$h(\lambda) =  \begin{vmatrix}
    \lambda I_{n-m} - A_{22}
    \end{vmatrix}$

\[
g(\lambda) = \begin{vmatrix}
\lambda & 0 & \cdots & 0 & -a_0 \\
-1 & \lambda & \cdots & 0 & -a_1 \\
0 & -1 & \cdots & 0 & -a_2 \\
\vdots & \vdots & \ddots & \vdots & \vdots \\
0 & 0 & \cdots & -1 & \lambda-a_{m-1}
\end{vmatrix} = \lambda^m - a_{m-1} \lambda^{m-1} - \cdots - a_1 \lambda - a_0
\]

   从而$g(\mathcal{A}) \xi = \mathcal{A}^m\xi - a_{m-1} \mathcal{A}^{m-1}\xi - \cdots - a_1 \mathcal{A}\xi - a_0 \xi = 0 $ (利用的是$\mathcal{A}^m\xi$的线性表示)

   由于同一变换的多项式具有可交换性，故
\[
f(\mathcal{A}) \xi = g(\mathcal{A}) h(\mathcal{A}) \xi = h(\mathcal{A}) g(\mathcal{A}) \xi  = h(\mathcal{A})0 = 0.
\]
    因此 $f(\mathcal{A}) = 0$。

\end{proof}

    \textbf{解法一（由 Vieta 定理）}
    \[
    \sum_{1\le i<j\le n}\lambda_i\lambda_j = \frac{1}{2}\left[\left(\sum_{i=1}^n\lambda_i\right)^2 - \sum_{i=1}^n\lambda_i^2\right].

    \[
    \sum_{1\le i<j\le n}\lambda_i\lambda_j = \frac{1}{2}\left[\left(\sum_{i=1}^n a_{ii}\right)^2 - \sum_{i=1}^n\sum_{j=1}^n a_{ij}a_{ji}\right].

考虑特征多项式

\[
\sigma_2 = \sum_{1\le i<j\le n}
\begin{vmatrix}
a_{ii} & a_{ij} \\
a_{ji} & a_{jj}
\end{vmatrix}

\]

\subsection{叙述、判断和说理题}

\begin{example}
    \begin{enumerate}
        \item[(1)] 线性映射的定义中的保加法和数乘这两个条件是否相互独立?
    
        \begin{solution}
            独立。保加法不能推出保数乘。反例：取复数域 \(\mathbb{C}\) 上的线性空间 \(V = W = \mathbb{C}\)，定义映射 \(f: \mathbb{C} \to \mathbb{C},\ f(z) = \overline{z}\)（共轭）。则
            \[
            f(z_1 + z_2) = \overline{z_1 + z_2} = \overline{z_1} + \overline{z_2} = f(z_1) + f(z_2),
            \]
            即保加法成立。但取 \(z = i,\ v = 1\)，有
            \[
            f(z \cdot v) = f(i) = -i,\quad z f(v) = i \cdot 1 = i,
            \]
            故 \(f(z v) \neq z f(v)\)，不满足保数乘。因此两个条件相互独立。
        \end{solution}
    
        \item[(2)] 请构造有限维线性空间 \(V\) 及 \(\phi \in \operatorname{End}V\) 分别满足以下条件的三个例子：
        \[
        \text{(a) } \ker \phi = \operatorname{Im} \phi;\qquad
        \text{(b) } \ker \phi + \operatorname{Im} \phi = V;\qquad
        \text{(c) } \ker \phi + \operatorname{Im} \phi \neq V.
        \]
    
        \begin{solution}
            取 \(\dim V = 2\)，设 \(\{e_1, e_2\}\) 为 \(V\) 的一组基。
    
            \textbf{(a)}  \(\phi(e_1,e_2) = (e_1,e_2)\begin{pmatrix}
                0 & 1 \\
                0 &  0\\
                \end{pmatrix}\) 即
            \[
            \phi(e_1) = 0,\quad \phi(e_2) = e_1.
            \]
            则 \(\operatorname{Im} \phi = Span(e_1)\)，\(ker \phi = Span(e_1)\)，故 \(ker \phi = \operatorname{Im} \phi\)。
    
            \textbf{(b)}  \(\phi(e_1,e_2) = (e_1,e_2) \begin{pmatrix}
                1 & 0 \\
                0 &  0\\
                \end{pmatrix}\) 即
            \[
            \phi(e_1) = e_1,\quad \phi(e_2) = 0.
            \]
            则 \(\operatorname{Im} \phi = Span(e_1)\)，\(ker \phi = Span(e_2)\)，且 \(ker \phi \cap \operatorname{Im} \phi = \{0\}\)，因此 \(ker \phi + \operatorname{Im} \phi = V\)。
    
            \textbf{(c)} 采用 (a) 中构造即满足 \(ker \phi + \operatorname{Im} \phi = Span(e_2) \neq V\)。
        \end{solution}
    
        \item[(3)] 
        \begin{enumerate}
            \item[(a)] 请给出 \(\phi\) 的特征值、特征向量和特征子空间的定义；
    
            \begin{solution}
                设 \(\phi \in \operatorname{End}V\)。若存在 \(\lambda \in P\) 和非零向量 \(\alpha \in V\) 使得
                \[
                \phi(\alpha) = \lambda \alpha,
                \]

                \]

            \end{solution}
    
            \item[(b)] 请给出无限维线性空间 \(V\) 上没有特征值的线性变换 \(\phi\) 的例子；
    
            \begin{solution}
                取 \(V = P[x]\) 为域 \(P\) 上全体多项式构成的无限维线性空间，定义

                \phi(f) = x f(x),\quad \forall f(x) \in P[x].
                \]
                若 \(\phi(f) = \lambda f\)，即 \(x f(x) = \lambda f(x)\)，则 \((x - \lambda) f(x) = 0\) 在多项式环中意味着 \(f(x) \equiv 0\)，故无非零特征向量。因此 \(\phi\) 没有特征值。

            \begin{solution}

\end{example}

    f(A) = \frac{2}{n} \operatorname{tr}(A) = \frac{2}{n} \sum_{i=1}^n a_{ii}.
    \]
    因为 \(a_{ii}\) 是 \(1,2,\dots,n\) 的一个全排列，所以 \(\sum_{i=1}^n a_{ii} =  \frac{n(n+1)}{2}\)，故
    \[
    f(A) = \frac{2}{n} \cdot \frac{n(n+1)}{2} = n+1.
    \]
\end{solution}

\begin{example}
    设$\phi$是$n$维复线性空间$V$上的线性变换.
    \begin{enumerate}
        \item[(1)]用几何方法证明 \(\phi\) 的 Jordan 标准形存在且唯一.
        \item[(2)]设$\lambda_0$是$\phi$的特征值，请分别给出$\phi$的 Jordan 标准形中属于$\lambda_0$的 Jordan 块的个数，属于$\lambda_0$的$k$阶 Jordan块的个数的几何意义，并说明理由.
    \end{enumerate}
\end{example}

\begin{proof}
        对于(1)先证明任意幂零变换 \(\phi\) 可将空间 \(V\) 分解为循环子空间的直和。对维数 \(n\) 进行归纳。
        
        当 \(n=1\) 时显然成立。设维数小于 \(n\) 时结论成立。对 \(n\) 维空间 \(V\) 上的幂零变换 \(\phi\)，存在正整数 \(k\le n\) 使得 \(\phi^k=0\) 且 \(\phi^{k-1}\neq0\)。取 \(\alpha\in V\) 使 \(\phi^{k-1}(\alpha)\neq0\)，则向量组 \(\alpha,\phi(\alpha),\dots,\phi^{k-1}(\alpha)\) 线性无关。令
        \[
        U = \operatorname{span}\{\alpha,\phi(\alpha),\dots,\phi^{k-1}(\alpha)\},
        \]
        则 \(U\) 是 \(\phi\)-不变子空间。若 \(U=V\)，则在这组基下 \(\phi\) 的表示矩阵即为 Jordan 块 \(J_k(0)\)，结论成立。
        
        若 \(U\neq V\)，设 \(W\) 是满足 \(W\cap U=0\) 的维数最大的 \(\phi\)-不变子空间（这样的 \(W\) 存在，例如取零子空间）。下证 \(V=U\oplus W\)。若不然，存在 \(\beta\in V\) 且 \(\beta\notin U\oplus W\)。由于 \(\phi^k=0\)，可设 \(\phi^t(\beta)\notin U\oplus W\) 而 \(\phi^{t+1}(\beta)\in U\oplus W\)（取最小的 \(t\)）。令 \(\gamma = \phi^t(\beta)\)，则 \(\phi(\gamma)\in U\oplus W\)。设 \(\phi(\gamma)=u+w\)，其中 \(u\in U,\ w\in W\)。由 \(\phi^k(\gamma)=0\) 得 \(\phi^{k-1}(u+w)=0\)，即 \(\phi^{k-1}(u)+\phi^{k-1}(w)=0\)。因为 \(\phi^{k-1}(w)\in W\)（\(W\) 不变）且 \(\phi^{k-1}(u)\in U\)，而 \(U\cap W=0\)，故 \(\phi^{k-1}(u)=0\) 且 \(\phi^{k-1}(w)=0\)。由 \(u\in U\) 可设 \(u = c_1\phi(\alpha)+\cdots+c_{k-1}\phi^{k-1}(\alpha)\)，则 \(u=\phi(x)\) 其中 \(x=c_1\alpha+\cdots+c_{k-1}\phi^{k-2}(\alpha)\in U\)。于是 \(w = \phi(\gamma)-u = \phi(\gamma-x)\)。由于 \(\gamma\notin U\oplus W\) 而 \(x\in U\)，故 \(\gamma-x\notin U\oplus W\)。令 \(W' = W \oplus \operatorname{span}(\gamma-x)\)，则 \(W'\) 仍是 \(\phi\)-不变子空间（因为 \(\phi(\gamma-x)=w\in W\)），且 \(\dim W'>\dim W\)，与 \(W\) 的极大性矛盾。因此 \(V=U\oplus W\)。由归纳假设 \(W\) 可分解为 \(\phi\)-循环子空间直和，从而 \(V\) 可分解为循环子空间直和，每个循环子空间对应一个 Jordan 块。
        
        对一般线性变换 \(\phi\)，设其特征值全在基域中（否则在代数闭包中考虑）。对每个特征值 \(\lambda\)，考虑根子空间 \(R_\lambda(\phi)=\ker(\phi-\lambda I)^n\)，则 \(V=\bigoplus R_\lambda(\phi)\)。在 \(R_\lambda(\phi)\) 上，\(\phi-\lambda I\) 是幂零的，故存在一组基使该限制的矩阵为 Jordan 形（对角元为 \(\lambda\)）。从而 \(\phi\) 的 Jordan 标准形存在。唯一性由根子空间的维数和每个 Jordan 块的大小（由 \(\dim\ker(\phi-\lambda I)^k\) 的跳跃决定）唯一确定。

    (2) 属于$\lambda_0$的Jordan块数：$n - r(\lambda_0I_V - \phi)$\\
        属于$\lambda_0$的$k$阶Jordan块数：$r(\lambda_0I_V - \phi)^{k+1} + r(\lambda_0I_V - \phi)^{k-1}- 2r(\lambda_0I_V - \phi)^k$\\
        几何意义：$k$维循环子空间的个数.
\end{proof}

\begin{example}
    设$\phi$是$n$维复线性空间$V$上的线性变换，$\phi$的特征多项式为$$f(\lambda) = \prod_{i = 1}^s(\lambda - \lambda_i)^{n_i},\;\lambda_i \neq \lambda_j ,1 \leq i \neq j \leq s $$
    我们知道有$V$关于$\phi$的根子空间分解$V = \bigoplus_{i=1}^s (\phi-\lambda_i id)^{n_i}$ \\
    称$R_{\lambda_i}(\phi) := ker(\phi-\lambda_i id)^{n_i} = \{\alpha \in V \mid (\phi - \lambda_i \mathrm{id})^{n_i}\alpha = 0\}.$为$\phi$的属于$\lambda_i$的根子空间，其中$id$是$V$上的恒等变换.\\
    证明：
    \begin{enumerate}
        \item[(1)]$dimR_{\lambda_i}(\phi) = n_i$;
        \item[(2)]$R_{\lambda_i}(\phi) = \bigcup_{k = 1}^{\infty} ker(\phi - \lambda_i id)^k$;
        \item[(3)] \((\phi - \lambda_i \mathrm{id})|_{R_{\lambda_j}(\phi)}\), \(\lambda_i \neq \lambda_j\)是可逆的；
        \item[(4)]若 \(\phi\) 的最小多项式为 \(d(\lambda) = \prod_{i=1}^{s}(\lambda - \lambda_i)^{m_i}\)，则 \(R_{\lambda_i}(\phi) = \ker (\phi - \lambda_i \mathrm{id})^{m_i}\).
    \end{enumerate}
\end{example}

\begin{proof}
    (1)
    \[
    V = \bigoplus_{i=1}^s R_{\lambda_i}(\phi).
    \]
    因此 \(n = \dim V = \sum_{i=1}^s \dim R_{\lambda_i}(\phi)\)。又因为 \(R_{\lambda_i}(\phi) = \ker (\phi-\lambda_i \mathrm{id})^{n_i}\)，而特征多项式与极小多项式有相同的根，且 \(f\) 的每个因子次数不小于 \(d\) 中对应因子的次数，所以 \(R_{\lambda_i}(\phi)\) 的维数不小于 \(n_i\)。因此 \(\dim R_{\lambda_i}(\phi) = n_i\)。

    (2)  \( A \overset{\text{def}}{=} \phi-\lambda_i I_V |_{R_{\lambda_i}} \) 则 \( \ker A \subset \ker A^2 \subset \cdots \subset \ker A^{n_i} \subset \cdots \)

    故 \( \dim \ker A \leq \cdots \leq \dim \ker A^{n_i} \leq \dim \ker A^{n_i+1} \leq \cdots \)
    
    则 \( r(A) \geq r(A^2) \geq \cdots \geq r(A^{n_i}) \geq r(A^{n_i+1}) \geq \cdots \)
    
    下证：\( \forall k \geq n_i, \quad r(A^k) = r(A^{k+1}) \)
    
    \[n_i \geq r(I_V|_{R_{\lambda_i(\phi)}}) \geq r(A^2) \geq r(A^{n_i+1}) \geq \cdots \geq r(A^{n_i}) \geq 0\]
    
    上面为 \( n_i+2 \) 个在 \( [0,n_i]\) 中的整数。故 \( \exists k_0 < n_i, s.t. \)
    
    \[r(A^{k_0}) = r(A^{k_0+1}), \quad \forall k \geq k_0\]
    
    \[r(A^{k_0+1}) = r(A^{k_0+1}, A^{n_i}) \geq r(A^{k_0}) + r(A^{n_i+1}) - r(A^{n_i})\]
    
    \[\Rightarrow r(A^{k_0+1}) \geq r(A^{k_0}) \quad \text{而显然} \quad r(A^{k_0}) \leq r(A^{k_0+1}), \quad \text{故} \quad \forall k \geq n_i, \quad r(A^{k_0}) = r(A^{k_0+1})\]
    
    \[\Rightarrow \dim \ker A^{k_0} = \dim \ker A^{k_0+1}, \quad \forall k \geq n_i\]
    
    从而
    
    \[\bigcup_{k=1}^{\infty} \ker(\phi-\lambda_i I_V)^{i}|_{R_{\lambda_i}} = \ker(\phi-\lambda_i I_V)^{n_i}|_{R_{\lambda_i}} = \ker(\phi-\lambda_i I_V)|_{R_{\lambda_i}}^{n_i}  = R_{\lambda_i}(\phi)\]

    (3)  \((\phi-\lambda_i I_v)_{R_{\lambda_j}}\) 的特征值为 \(\lambda_i - \lambda_j \neq 0\)。故可逆。

    (4) 易知 \( d(\phi) =  \prod_{i= 1}^s (\phi - \lambda_i I_V)^{m_i} \)，令 \( g(x) = \frac{d(x)}{(x-\lambda_k)^{m_k}} \)，易知$m_k \leq n_k$

    \( \forall d \in V, \quad g_k(\phi)(\alpha) \in \ker (\phi - \lambda_i I_V)^{m_i} \)，因为 \((\phi - \lambda_i I_V)^{m_i} g_k(e)\alpha = 0\cdot a = 0\)
    
    \((g_1, g_2, \cdots, g_s) = 1\)，故 \(\exists u_1, \cdots, u_s \in \mathbb{C}[x]\)，\( s.t. \quad u_1(x) g_1(x) + \cdots + u_s(x) g_s(x) = 1\)
    
    即 \( u_1(\phi)g_1(\phi) + \cdots + u_s(\phi)g_s(\phi) = I_V \)
    
    故 \( \forall \alpha \in V, \quad \alpha = u_1(\phi)g_1(\phi)(\alpha) + \cdots + u_s(\phi)g_s(\phi)(\alpha) \)，故 \( V = \ker(\varphi - \lambda_1 I_V)^{m_1} + \cdots + \ker(\varphi - \lambda_s I_V)^{m_s} \)
    
    再证：  
    \[ V = \ker(\varphi - \lambda_1 I_V)^{m_1} + \cdots + \ker(\varphi - \lambda_s I_V)^{m_s} \] 是直和  
    
    \[\forall \beta \in \sum_{k\neq j} \ker(\varphi - \lambda_k I_V)^{m_k} \cap \ker(\varphi - \lambda_j I_V)^{m_j} \]
    
    \[\beta = \beta_j = \sum_{k \neq j} \beta_k, \quad \beta = \beta \cdot I_V = \sum_{k\neq j} u_k(\phi)g_k(\phi)(\beta) + u_j(\phi) g_j(\phi)(\beta_j) = 0\]
    
    \[\Rightarrow \beta = 0, \quad \text{故} \quad V = \bigoplus_{k=1}^s (\varphi - \lambda_k I_V)^{m_k} = \bigoplus_{k=1}^s (\varphi - \lambda_k I_V)^{n_k}, \quad m_k \leq n_k\]
    
    而 \(\ker(\varphi - \lambda_k I_V)^{m_k} \subset \ker(\varphi - \lambda_k I_V)^{n_k}\), 故 \(R_{\lambda_i}(\varphi) = \ker(\varphi - \lambda_k I_V)^{n_k} = \ker(\varphi - \lambda_k I_V)^{m_k}\)
\end{proof}

\begin{example}
    设 \(V\) 是数域 \(P\) 上有限维线性空间，\(\phi \in \operatorname{End}V\)，\(f(\lambda)\) 是 \(\phi\) 的特征多项式。证明：\(f(\lambda)\) 在 \(P\) 上可约当且仅当 \(\phi\) 有非平凡的不变子空间。
\end{example}

\begin{proof}
    “\(\Leftarrow\)”：若 \(\phi\) 有非平凡的不变子空间 \(U\)，取 \(U\) 的一组基扩充为 \(V\) 的基，则 \(\phi\) 在此基下的矩阵为分块上三角阵：
    \[
    \begin{pmatrix} A & B \\ 0 & C \end{pmatrix},
    \]
    其特征多项式 \(f(\lambda)=\det(\lambda I-A)\cdot\det(\lambda I-C)\) 为两个次数均小于 \(\dim V\) 的多项式之积，故 \(f\) 可约。
    
    “\(\Rightarrow\)”：设 \(f(\lambda)=g(\lambda)h(\lambda)\)，其中 \(\deg g,\deg h\ge 1\)。由 Hamilton-Cayley 定理，\(f(\phi)=0\)，故 \(g(\phi)h(\phi)=0\)。因为 \(g(\phi)h(\phi)=0\)，若 \(g(\phi)\) 和 \(h(\phi)\) 均非零，则至少有一个是奇异的（否则两边左乘逆得 \(h(\phi)=0\) 或 \(g(\phi)=0\)）。不妨设 \(g(\phi)\) 奇异，则存在非零 \(\alpha\in \ker g(\phi)\)。考虑由 \(\alpha\) 生成的循环子空间 \(C(\alpha,\phi)=\operatorname{span}\{\alpha,\phi(\alpha),\dots,\phi^{r-1}(\alpha)\}\)，其中 \(r=\deg g\)。由于 \(g(\phi)\alpha=0\)，该子空间维数不超过 \(r\)，且显然非零。若它等于整个 \(V\)，则 \(\dim V\le r\)，但 \(\deg f=n\)，\(f=gh\)，\(r\le n\)，若 \(r=n\) 则 \(h\) 为常数，\(f\) 不可约，矛盾。故该子空间为非平凡的不变子空间。
\end{proof}

\begin{example}
    设 \(V\) 是数域 \(P\) 上 \(n\) 维线性空间，\(\phi \in \operatorname{End}V\)，称 \(V\) 是 \(\phi\) 的循环空间，如果存在 \(\alpha \in V\)，使 \(\alpha, \phi\alpha, \ldots, \phi^{n-1}\alpha\) 是 \(V\) 的一组基。证明：\(V\) 是 \(\phi\) 的循环空间当且仅当 \(\phi\) 的特征多项式等于 \(\phi\) 的最小多项式。
\end{example}

\begin{proof}
    “\(\Rightarrow\)”：设 \(V\) 是循环空间，取基 \(\alpha,\phi\alpha,\dots,\phi^{n-1}\alpha\)，则 \(\phi\) 在此基下的矩阵为 Frobenius 块：
    \[
    \begin{pmatrix}
    0 & 0 & \cdots & 0 & -a_0 \\
    1 & 0 & \cdots & 0 & -a_1 \\
    0 & 1 & \cdots & 0 & -a_2 \\
    \vdots & \vdots & \ddots & \vdots & \vdots \\
    0 & 0 & \cdots & 1 & -a_{n-1}
    \end{pmatrix},
    \]
    其特征多项式为 \(\lambda^n + a_{n-1}\lambda^{n-1}+\cdots+a_0\)，且该矩阵的最小多项式等于特征多项式（因为有理标准形中只有一个块）。故 \(\phi\) 的特征多项式等于最小多项式。
    
    “\(\Leftarrow\)”：若特征多项式 \(f(\lambda)\) 等于最小多项式，则 \(\phi\) 的有理标准形只有一个 Frobenius 块，即存在一组基 \(e_1,\dots,e_n\) 使得
    \[
    \phi(e_1)=e_2,\ \phi(e_2)=e_3,\ \dots,\ \phi(e_{n-1})=e_n,\ \phi(e_n)=-a_0e_1-a_1e_2-\cdots-a_{n-1}e_n.
    \]
    于是 \(e_k = \phi^{k-1}(e_1)\)，从而 \(e_1,\phi(e_1),\dots,\phi^{n-1}(e_1)\) 构成一组基，故 \(V\) 是循环空间。
\end{proof}
